```dart
//Scaffold用于构建上中下页面布局
//1. appBar                 头部区域
//2. body                   中部区域
//3. bottomNavigationBar    底部区域
//4. backgroundColor        Scaffold的背景色
//5. floatingActionButton   悬浮操作按钮，常用于触发页面的主要动作
//6. ...
//上述属性中的子属性
//child表示控件中的子元素
//center表示居中
//Text用于展示文字
//height控制高度

import 'package:flutter/material.dart';

void main() {
  runApp(MaterialApp(
    title: "Flutter组件初体验",
    theme: ThemeData(scaffoldBackgroundColor: Colors.blue),
    home: Scaffold(
      appBar: AppBar(
        title: Text("top bar/appbar"),
      ),
      body: Container(
        child: Center(
          child: Text("content/body"),
        ),
      ),
      bottomNavigationBar: Container(
        height: 80,//不设置高度会直接覆盖整个中部布局
        child: Center(
          child: Text("footer/bottomNavigationBar"),
        ),
      ),
    ),
  ));
}
```
