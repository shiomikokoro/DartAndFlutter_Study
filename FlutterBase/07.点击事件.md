```dart
import 'package:flutter/material.dart';

void main() {
  runApp(MainPage());
}
class MainPage extends StatefulWidget {
  @override
  State<StatefulWidget> createState() => _MainPageState();
}

class _MainPageState extends State<MainPage> {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: "有状态组件",
      home: Scaffold(
        //topbar
        appBar: AppBar(
          title: Text("top bar/appbar"),
        ),
        //content
        body: Container(
          child: Center(
            //GestureDetector 手势监听，也是Flutter中最常用、功能最丰富的手势监听组件
            //用户与应用程序交互触发的动作，比如点击、滑动、触摸等
            //用改类包裹对应元素
            child: GestureDetector(//手势监听
              onTap:() => print("thanks for click"),//点击事件
              onDoubleTap: () => print("thanks for dounle click"),//双击事件
              onLongPress: () => print("wow, you hold me~"),//长按事件
              child: Text("content/body"),
            ),
          ),
        ),
        //footer
        bottomNavigationBar: Container(
          height: 80,
          child: Center(
            //还有一些专门用于触发点击事件的按钮类
            //比如调用onPressed参数的专用组件
            ////TextButton、ElevatedButton、OutLineButtton、FloatingActionButton
            //比如调用OnTap，还有水纹扩散风格效果的MaterialDesign
            ////InkWell
            //比如具有特定功能的交互式控件，用onPressed触发
            ////IconButton、Switch、Checkbox
            child: TextButton(//专用点击事件的按钮，自带动画等
              onPressed: ()=>print("button click"),//被点击的时候
              child: Text("footer/bottomNavigationBar"),//按钮文本
            ),
          ),
        ),
      ),
    );
  }

}
```
